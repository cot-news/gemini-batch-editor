async function processSingleImage(file, key) { const formData = new FormData(); formData.append('file', file); // 如果之后要集成 Gemini 处理，可以在这里传 key formData.append('api_key', key); // 将此处替换为你从 Hugging Face 获取的 Direct URL + /process const response = await fetch('https://naidd-my-ai-remover.hf.space/process', { method: 'POST', body: formData }); if (!response.ok) throw new Error("服务器处理失败"); const blob = await response.blob(); return URL.createObjectURL(blob); }
